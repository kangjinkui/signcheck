| 항목 | 내용 |

### Value Delivered (4관점)

| 관점 | 내용 |
|------|------|
| **Problem** | 담당자 수동 법령 검색 → 민원 응대 지연 및 비표준 답변 |
| **Solution** | 규칙 엔진(판정) + PrivateGPT Docker(RAG) 분리 구조로 완전 오프라인 자동 판정 |
| **Function UX** | 구조화 폼 → 1초 판정 + 조문 근거 → 챗봇 추가 질문 |
| **Core Value** | 동일 입력 동일 결과 보장, 내부망 완전 오프라인 운용 |

```sql
-- pgvector 확장 활성화
CREATE EXTENSION IF NOT EXISTS vector;

-- 법령 문서 마스터
CREATE TABLE document_master (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name           VARCHAR(200) NOT NULL,
  type           VARCHAR(50),        -- 'law', 'ordinance', 'guideline'
  jurisdiction   VARCHAR(50),        -- 'national', 'seoul', 'gangnam'
  effective_date DATE,
  version        VARCHAR(20),
  source_type    VARCHAR(20),        -- 'api', 'pdf'
  file_url       TEXT,
  created_at     TIMESTAMP DEFAULT NOW()
);

-- 조문 원문
CREATE TABLE provision (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  document_id    UUID REFERENCES document_master(id),
  article        VARCHAR(50),        -- 제3조
  paragraph      VARCHAR(50),        -- 제1항
  item           VARCHAR(50),        -- 제1호
  content        TEXT NOT NULL,
  effective_date DATE
);

-- 조문 관계
CREATE TABLE legal_relation (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  from_provision UUID REFERENCES provision(id),
  to_provision   UUID REFERENCES provision(id),
  relation_type  VARCHAR(50)         -- 'applies', 'overrides', 'exception'
);

-- 판정 규칙 조건
CREATE TABLE rule_condition (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  sign_type      VARCHAR(100) NOT NULL,  -- 광고물 유형
  floor_min      INT,
  floor_max      INT,
  light_type     VARCHAR(50),            -- 'none', 'internal', 'neon_digital'
  digital        BOOLEAN DEFAULT FALSE,
  area_min       DECIMAL(10,2),
  area_max       DECIMAL(10,2),
  zone           VARCHAR(100),           -- 용도지역
  ad_type        VARCHAR(20),            -- 'self', 'third_party'
  tehranro       BOOLEAN DEFAULT FALSE,  -- 테헤란로 접면 여부
  vendor_count_min INT                   -- 지주이용간판 업체 수
);

-- 판정 규칙 결과
CREATE TABLE rule_effect (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  rule_id        UUID REFERENCES rule_condition(id),
  decision       VARCHAR(20) NOT NULL,   -- 'permit', 'report', 'prohibited'
  review_type    VARCHAR(50),            -- '본심의', '소심의', '서울시심의'
  safety_check   BOOLEAN DEFAULT FALSE,
  max_area       DECIMAL(10,2),
  max_height     DECIMAL(10,2),
  max_protrusion DECIMAL(10,2),
  display_period VARCHAR(20),            -- '3년'
  provision_id   UUID REFERENCES provision(id)
);

-- 수수료 규칙
CREATE TABLE fee_rule (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  sign_type      VARCHAR(100) NOT NULL,
  area_threshold DECIMAL(10,2),         -- 기준 면적 (예: 5)
  base_fee       INT NOT NULL,          -- 기준 수수료
  extra_fee      INT,                   -- 초과 1㎡당 추가 수수료
  light_weight   DECIMAL(5,2) DEFAULT 1.0,  -- 조명 가중치
  digital_weight DECIMAL(5,2) DEFAULT 2.0,  -- 디지털 가중치
  provision_id   UUID REFERENCES provision(id)
);

-- 서류 체크리스트 규칙
CREATE TABLE checklist_rule (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  work_type      VARCHAR(50),           -- 'permit', 'report', 'change', 'extend'
  required_docs  JSONB,                 -- 필수 서류 목록
  optional_docs  JSONB,                 -- 선택 서류 목록
  condition      TEXT
);

-- 용도지역 규칙
CREATE TABLE zone_rule (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name           VARCHAR(100) NOT NULL,
  restriction    TEXT,
  prohibited_types JSONB,              -- 금지 광고물 유형 목록

);

-- 판정 이력 로그
CREATE TABLE case_log (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  input_data       JSONB NOT NULL,
  output_data      JSONB NOT NULL,
  rule_version     VARCHAR(20),
  decision         VARCHAR(20),
  fee_calculated   INT,
  rag_chunks_used  JSONB,              -- 참조된 조문 청크 ID 목록
  user_id          VARCHAR(100),
  created_at       TIMESTAMP DEFAULT NOW()
);

# backend/services/privategpt_client.py
import httpx

PRIVATEGPT_URL = "http://privategpt:8080"

async def query_rag(question: str, context_filter: dict = None) -> dict:
    """판정 근거 조문 검색"""
    async with httpx.AsyncClient() as client:
        response = await client.post(
            f"{PRIVATEGPT_URL}/v1/completions",
            json={
                "prompt": question,
                "use_context": True,
                "context_filter": context_filter,
                "include_sources": True,
                "stream": False
            },
            timeout=30.0
        )
    return response.json()

async def ingest_document(file_path: str) -> dict:
    """법령 문서 임베딩"""
    async with httpx.AsyncClient() as client:
        with open(file_path, 'rb') as f:
            response = await client.post(
                f"{PRIVATEGPT_URL}/v1/ingest/file",
                files={"file": f},
                timeout=120.0
            )
    return response.json()
```

## 7. 규칙 엔진 설계

```python
# backend/engine/rule_engine.py

class RuleEngine:
    def judge(self, input: JudgeInput) -> JudgeResult:
        # 1. 절대 불가 체크 (최우선)
        if self._check_prohibited(input):
            return JudgeResult(decision="prohibited", ...)

        # 2. 규격 초과 체크
        spec_check = self._check_spec(input)
        if spec_check.exceeded:
            return JudgeResult(decision="prohibited", warnings=spec_check.warnings)

        # 3. 허가 vs 신고 분기
        decision = self._check_permit_or_report(input)

        # 4. 심의 분류
        review_type = self._classify_review(input, decision)

        # 5. 수수료 계산
        fee = FeeCalculator().calculate(input)

        # 6. 서류 목록 생성
        docs = ChecklistGenerator().generate(input, decision)

        return JudgeResult(
            decision=decision,
            review_type=review_type,
            fee=fee,
            required_docs=docs
        )

    def _check_prohibited(self, input) -> bool:
        """절대 불가 조건"""
        # 테헤란로 + 돌출간판
        if input.tehranro and input.sign_type == "돌출간판":
            return True
        # 용도지역별 금지 유형
        zone_rule = db.query(ZoneRule).filter_by(name=input.zone).first()
        if zone_rule and input.sign_type in zone_rule.prohibited_types:
            return True
        return False
```

## 8. 수수료 계산 로직

```python
# backend/engine/fee_calculator.py

FEE_TABLE = {
    "벽면이용간판": {
        "self":  {"base_area": 5, "base_fee": 4000,  "extra": 1500},
        "third": {"base_area": 5, "base_fee": 40000, "extra": 2000},
    },
    "돌출간판": {
        "self":  {"base_area": 5, "base_fee": 20000, "extra": 2000},
    },
    "옥상간판": {
        "self":  {"base_area": 10, "base_fee": 40000, "extra": 6000},
    },
    # ... 13종 전체
}

LIGHT_WEIGHT = {"none": 1.0, "internal": 1.5, "neon_digital": 2.0}

def calculate(input: JudgeInput) -> FeeResult:
    rule = FEE_TABLE[input.sign_type][input.ad_type]
    base = rule["base_fee"]
    if input.area > rule["base_area"]:
        base += (input.area - rule["base_area"]) * rule["extra"]
    weight = LIGHT_WEIGHT[input.light_type]
    total = round(base * weight / 100) * 100  # 100원 단위 반올림
    return FeeResult(base=int(base), weight=weight, total=total)
```

## 10. 개발 순서 (단계별 체크리스트)

### 1단계: PostgreSQL 스키마 + 법령 데이터 수집
- [ ] `init.sql` 작성 (전체 테이블 생성)

- [ ] 수수료 테이블 데이터 입력 (PDF 기준)
- [ ] 규칙 조건/결과 테이블 초기 데이터 입력

### 2단계: PrivateGPT Docker 설정 + 임베딩
- [ ] `docker-compose.yml` 작성
- [ ] Ollama 모델 사전 다운로드 (llama3.1, nomic-embed-text)
- [ ] PrivateGPT `settings.yaml` 설정
- [ ] 법령 문서 ingest (텍스트 파일로 변환 후 `/v1/ingest/file`)
- [ ] RAG 검색 동작 확인 (`/v1/completions` 테스트)

### 3단계: FastAPI 규칙 엔진
- [ ] `rule_engine.py` - 판정 로직 13종 전체
- [ ] `fee_calculator.py` - 수수료 계산 (PDF 수수료 표 기준)
- [ ] `checklist.py` - 서류 목록 생성
- [ ] `POST /api/v1/judge` 엔드포인트 연동 테스트

### 4단계: PrivateGPT 연동 + 통합 API
- [ ] `privategpt_client.py` 작성
- [ ] 판정 결과에 RAG 근거 조문 붙이기

### 5단계: Next.js UI

## 11. 내부망 오프라인 배포 패키지

```bash
# 외부망에서 사전 준비 (배포 전)
ollama pull llama3.1:8b-instruct-q4_K_M
ollama pull nomic-embed-text
docker save privategpt ollama postgres > adjudge-images.tar
pip download -r requirements.txt -d ./pip-packages/
npm pack  # node_modules 사전 설치

# 내부망 이전 후
docker load < adjudge-images.tar
pip install --no-index --find-links=./pip-packages/ -r requirements.txt
docker-compose up -d
```

## 12. 비기능 요구사항 달성 방법

| 요구사항 | 달성 방법 |
|---------|-----------|
| 판정 1초 이내 | 규칙 엔진 PostgreSQL 인덱스 + 메모리 캐시 |
| RAG 3초 이내 | pgvector IVFFlat 인덱스, top_k=3 제한 |
| 동일 입력 → 동일 결과 | 규칙 엔진 100% 결정론적 (LLM 판정 없음) |
| 전체 로그 저장 | case_log 테이블, 모든 입출력 JSON 저장 |
| 오프라인 완전 운용 | Docker Compose 자급자족, 외부 API 없음 |
